Examples

Start writing a function for coefficient of variation

cv <- function(var) {
    sd(var) / mean(var)
}
x=3 ; cv(c(3,x))
## [1] 0
x=6 ; cv(c(3,x))
## [1] 0.4714045

First example. Capture and uncapture

cv <- function(var) {
    var <- enexpr(var)
    expr(sd(!!var) / mean(!!var))
}
x=3
eval(cv(c(3,x)))
## [1] 0

#Second example. This works even if we put x in the enviroment where the finction is written bc is not going to be used.

cv <- function(var) {
    x=6
    var <- enexpr(var)
    expr(sd(!!var) / mean(!!var))
}
x=3
eval(cv(c(3,x)))
## [1] 0

#Third example. That doesnt work the way we would like if we are modifying a df

adding_cv <- function(df,var) {
    x=c(3,3)
    var <- enexpr(var)
    mutate(df, sd(!!var) / mean(!!var))
}

x=c(3,6)
df<- tibble(n=3)
adding_cv(df,x)
## # A tibble: 1 x 2
##       n `sd(x)/mean(x)`
##   <dbl>           <dbl>
## 1     3               0

#Fourth example. To make it work the way we want, capture it with enquo

adding_cv <- function(df,var) {
    x=c(3,3)
    var <- enquo(var)
    mutate(df, sd(!!var) / mean(!!var))
}

x=c(3,6)
df<- tibble(n=3)
adding_cv(df,x)
## # A tibble: 1 x 2
##       n `sd(x)/mean(x)`
##   <dbl>           <dbl>
## 1     3           0.471

#Firth example. Let’s make that a prettier name

adding_cv <- function(df,var,nm) {
    x=c(3,3)
    nm_name <- quo_name(nm)
    var <- enquo(var)
    mutate(df, !!nm_name:= sd(!!var) / mean(!!var))
}

x=c(3,6)
df<- tibble(n=3)
adding_cv(df,x,"pretty_name")
## # A tibble: 1 x 2
##       n pretty_name
##   <dbl>       <dbl>
## 1     3       0.471

#Sixth example. Insert a list of expressions into a call

adding_cv <- function(df,var,nm, ...) {
    x=c(3,3)
    nm_name <- quo_name(nm)
    var <- enquo(var)
    filtering <- quos(...)
    df %>%  filter(!!!filtering) %>% 
    mutate( !!nm_name:= sd(!!var) / mean(!!var))}
x=c(3,6)
df<- tibble(n=c(3,6,9),m=c(3,6,9))
adding_cv(df,x,"pretty_name", n %in% c(3,6))
## # A tibble: 2 x 3
##       n     m pretty_name
##   <dbl> <dbl>       <dbl>
## 1     3     3       0.471
## 2     6     6       0.471

Conclusions
